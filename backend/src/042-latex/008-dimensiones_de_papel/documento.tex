% Ajustar por medidas manualmente con bordes:
\usepackage[papersize={210mm,297mm},lmargin=1mm,rmargin=1mm,top=1mm,bottom=1mm]{geometry}

% También podemos importar el paquete y colocar después las medidas con el comando \geometry
\usepackage{geometry}
\geometry{a4paper,total={170mm,257mm},left=20mm,top=20mm}
